Enthalpy Equations\\
\begin{align*}
\begin{split}
\
\Delta H &= \Delta U + P \Delta V \\ 
\
\Delta H &= \Delta U + \Delta n RT\\
\
\Delta H &= \frac{C_p}{\Delta T}\\
\
\Delta U &= \frac{C_V}{\Delta T}\\
\
C_p &= C_v + nR\\
\
w_\text{sys} & = - \int_{V_f}^{V_i} P dV
\end{split}
\end{align*}
